
CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
CMAKE_POLICY(SET CMP0007 NEW)

PROJECT(bytebuf C)
SET(bytebuf_VERSION_MAJOR 1)
SET(bytebuf_VERSION_MINOR 0)
SET(bytebuf_VERSION_PATCH 0)
SET(bytebuf_VERSION_TWEAK 0)

SET(TMP_PROJECT_DESCRIPTION "Simple circular buffer for embedded systems"
        CACHE STRING "Project Description")

SET(CMAKE_INSTALL_PREFIX ${PLATFORM_PACKAGES_PATH})
SET(PUBLIC_INCLUDE_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
        
FILE(GLOB SOURCES *.c)
FILE(GLOB_RECURSE HEADERS *.h)
ADD_PLATFORM_LIBRARY(bytebuf STATIC "" ${SOURCES})
INSTALL_PLATFORM_LIBRARY(bytebuf ${HEADERS})
